<textarea id="tarefaTexto" rows="4" cols="30" placeholder="Digite sua tarefa..."></textarea><br><br>

<button onclick="salvarTarefa()">Salvar</button>

<script>

const tarefas = {};

const meses = ["Janeiro", "Fevereiro", "Março", "Abril", "Maio", "Junho", "Julho", "Agosto", "Setembro", "Outubro", "Novembro", "Dezembro"];


function criarCalendario() {

const container = document.getElementById("calendarioContainer"); for (let mes = 0; mes < 12; mes++) {

const data = new Date(2025, mes, 1);

const mesDiv = document.createElement("div"); mesDiv.classList.add("mes");

const titulo = document.createElement("h2"); titulo.textContent = meses[mes]; mesDiv.appendChild(titulo);


const diasDiv = document.createElement("div"); diasDiv.classList.add("dias");


const primeiroDiaSemana = data.getDay();

const diasNoMes = new Date(2025, mes + 1, 0).getDate();


for (let i = 0; i < primeiroDiaSemana; i++) { const vazio = document.createElement("div"); diasDiv.appendChild(vazio);

}


for (let dia = 1; dia <= diasNoMes; dia++) {

const diaDiv = document.createElement("div"); diaDiv.classList.add("dia");

diaDiv.textContent = dia;

diaDiv.onclick = () => abrirPopup(mes, dia); diasDiv.appendChild(diaDiv);

}


mesDiv.appendChild(diasDiv); container.appendChild(mesDiv);

}

}


function abrirPopup(mes, dia) {

const popup = document.getElementById("popup");

const data = `${dia.toString().padStart(2, '0')}/${(mes+1).toString().padStart(2, '0')}/2025`;

document.getElementById("dataSelecionada").textContent = data; document.getElementById("tarefaTexto").value = ""; popup.style.display = "block";

}


function fecharPopup() { document.getElementById("popup").style.display = "none";

}


function salvarTarefa() {

const data = document.getElementById("dataSelecionada").textContent; const texto = document.getElementById("tarefaTexto").value.trim();

if (!texto) return;


if (!tarefas[data]) tarefas[data] = []; tarefas[data].push(texto); fecharPopup(); atualizarListaTarefas();

}


function atualizarListaTarefas() {

const container = document.getElementById("listaTarefas"); container.innerHTML = "";

for (const data in tarefas) {

const bloco = document.createElement("div"); bloco.classList.add("tarefa-dia");

bloco.innerHTML = `<h4>${data}</h4><ul>` + tarefas[data].map(t => `<li contenteditable="true">${t}</li>`).join("") + "</ul>";

container.appendChild(bloco);

}

}

window.onload = criarCalendario;

</script>
